432 bJSON_MAKE_CONSTEXPR_SERIALIZABLE(JSONValue::LiteralType)
433 {
434 std::string serialized{""};
435 switch (val)
436 {
437 case JSONValue::LiteralType::null_v:
438 serialized.append("null");
439 break;
440 case JSONValue::LiteralType::true_v:
441 serialized.append("true");
442 break;
443 case JSONValue::LiteralType::false_v:
444 serialized.append("false");
445 break;
446 default:
447 throw std::exception{"JSONLiteral had invalid value -- was it empty initialized by accident?"};
448 break;
449 }
450 return serialized;
451 }
452
453 bJSON_MAKE_SERIALIZABLE(JSONValue::NumberType)
454 {
455 std::string serialized{std::format("{}", val)};
456 return serialized;
457 }
458
459 bJSON_MAKE_SERIALIZABLE(JSONValue::StringType)
460 {
461 std::string serialized{val};
462
463 serialized = std::regex_replace(serialized, std::regex{"\\\\"}, "\\\\");
464 serialized = std::regex_replace(serialized, std::regex{"\""}, "\\\"");
465 serialized = std::regex_replace(serialized, std::regex{"\n"}, "\\n");
466 serialized = std::regex_replace(serialized, std::regex{"\r"}, "\\r");
467 serialized = std::regex_replace(serialized, std::regex{"\t"}, "\\t");
468 serialized = std::regex_replace(serialized, std::regex{"\f"}, "\\f");
469 serialized = std::regex_replace(serialized, std::regex{"\b"}, "\\b");
470
471 serialized.insert(serialized.cbegin(), '"');
472 serialized.insert(serialized.cend(), '"');
473 return serialized;
474 }
475
476 bJSON_MAKE_CONSTEXPR_SERIALIZABLE(JSONValue::ArrayType)
477 {
478 std::string serialized{"["};
479 for (auto first{true}; const auto &element : val)
480 {
481 if (element.type == JSONValue::JSONValueType::undefined)
482 {
483 continue;
484 }
485
486 if (!first)
487 {
488 serialized.append(",");
489 }
490 else
491 {
492 first = false;
493 }
494
495 serialized.append(" ");
496 serialized.append(serialize(element));
497 }
498 serialized.append(" ]");
499 return serialized;
500 };
501
502 bJSON_MAKE_SERIALIZABLE(JSONValue::ObjectType)
503 {
504 std::string serialized{"{"};
505 for (auto first{true}; const auto &[key, value] : val)
506 {
507 if (value.type == JSONValue::JSONValueType::undefined)
508 {
509 continue;
510 }
511
512 if (!first)
513 {
514 serialized.append(",");
515 }
516 else
517 {
518 first = false;
519 }
520
521 serialized.append(" ");
522 serialized.append(serialize(JSONValue{key}));
523 serialized.append(" : ");
524 serialized.append(serialize(value));
525 }
526 serialized.append(" }");
527 return serialized;
528 }

530 bJSON_DEFINE_CONSTEXPR_SERIALIZATION(JSONValue)
531 {
532 std::string serialized{""};
533 switch (val.type)
534 {
535 case JSONValue::JSONValueType::literal:
536 serialized.append(serialize(std::get<JSONValue::LiteralType>(val.value)));
537 break;
538 case JSONValue::JSONValueType::number:
539 serialized.append(serialize(std::get<JSONValue::NumberType>(val.value)));
540 break;
541 case JSONValue::JSONValueType::string:
542 serialized.append(serialize(std::get<JSONValue::StringType>(val.value)));
543 break;
544 case JSONValue::JSONValueType::array:
545 serialized.append(serialize(std::get<JSONValue::ArrayType>(val.value)));
546 break;
547 case JSONValue::JSONValueType::object:
548 serialized.append(serialize(std::get<JSONValue::ObjectType>(val.value)));
549 break;
550 case JSONValue::JSONValueType::undefined:
551 default:
552 throw std::exception{"JSONValue::type was 'undefined' -- it can not be serialized!"};
553 break;
554 }
555 return serialized;
556 }

716/* //
717// DO NOT REMOVE THIS SECTION! //
718// //
719// See the above macro usages for examples of how to serialize a type. //
720// //
721// As an additional example, consider the hypothetical struct which has the following definition: //
722// //
723// struct Example //
724// { //
725// const std::string name{""}; //
726// Example* const parent{nullptr}; //
727// }; //
728// //
729// To declare Example as json serializable and to define the serialization implementation, the helper macro is used: //
730// //
731// bJSON_MAKE_SERIALIZABLE(Example) //
732// { //
733// if (val.name.empty()) //
734// { //
735// throw std::exception{"Example::name was empty-- cannot serialize to JSON."}; //
736// } //
737// std::string serialized{R"""({ "name" : )"""}; //
738// serialized.append(serialize(val.name)); //
739// serialized.append(R"""(, "parent" : )"""); //
740// if (val.parent) //
741// { //
742// serialized.append(serialize(val.parent->name)); //
743// } //
744// else //
745// { //
746// serialized.append(serialize(JSONValue::LiteralType::null_v)); //
747// } //
748// serialized.append(" }"); //
749// return serialized; //
750// } //
