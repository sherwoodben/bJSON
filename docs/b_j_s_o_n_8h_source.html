407 bJSON_MAKE_SERIALIZABLE_INLINE(JSONValue::LiteralType)
408 {
409 std::u8string serialized{u8""};
410 switch (val)
411 {
412 case JSONValue::LiteralType::null_v:
413 serialized.append(u8"null");
414 break;
415 case JSONValue::LiteralType::true_v:
416 serialized.append(u8"true");
417 break;
418 case JSONValue::LiteralType::false_v:
419 serialized.append(u8"false");
420 break;
421 default:
422 throw std::exception{"JSONLiteral had invalid value -- was it empty initialized by accident?"};
423 break;
424 }
425 return serialized;
426 }
427
428 bJSON_MAKE_SERIALIZABLE_INLINE(JSONValue::NumberType)
429 {
430 std::array<char, std::numeric_limits<JSONValue::NumberType>::max_digits10 + 1> ascii{'\0'};
431
432 // convert the number to characters...
433 std::to_chars_result res = std::to_chars(ascii.data(), ascii.data() + ascii.size(), val);
434 if (res.ec != std::errc{})
435 {
436 throw std::exception{std::make_error_code(res.ec).message().c_str()};
437 }
438
439 // avoid UB with reinterpret_cast... annoying but quick and dirty for now.
440 std::u8string utf8;
441 for (const auto c : ascii)
442 {
443 if (c == '\0')
444 {
445 break;
446 }
447
448 utf8.push_back(static_cast<char8_t>(c));
449 }
450
451 return utf8;
452 }
453
454 bJSON_MAKE_SERIALIZABLE_INLINE(JSONValue::StringType)
455 {
456 std::u8string serialized{u8"\""};
457
458 // avoid UB with reinterpret_csat... but also avoid regex since it would lead to issues with unicode
459 // characters! Also be sure to escape any of the "control characters" as per the JSON specification
460
461 for (const auto &unit : val)
462 {
463
464 switch (unit)
465 {
466 case u8'\\':
467 serialized.append(u8"\\\\");
468 continue;
469 case u8'\"':
470 serialized.append(u8"\\\"");
471 continue;
472 case u8'\n':
473 serialized.append(u8"\\n");
474 continue;
475 case u8'\r':
476 serialized.append(u8"\\r");
477 continue;
478 case u8'\t':
479 serialized.append(u8"\\t");
480 continue;
481 case u8'\f':
482 serialized.append(u8"\\f");
483 continue;
484 case u8'\b':
485 serialized.append(u8"\\b");
486 continue;
487 default:
488 break;
489 }
490 if (unit >= 0x00 && unit <= 0x1F)
491 {
492 std::array<char, 3> asHex{'\0'};
493 std::to_chars_result res = std::to_chars(asHex.data(), asHex.data() + asHex.size(), unit, 16);
494 if (res.ec != std::errc{})
495 {
496 throw std::exception{std::make_error_code(res.ec).message().c_str()};
497 }
498 serialized.append(u8"\\u");
499 size_t length{strnlen(asHex.data(), 2)};
500 for (size_t i = 0; i < 4 - length; ++i)
501 {
502 serialized.push_back(u8'0');
503 }
504 for (size_t i = 0; i < length; ++i)
505 {
506 serialized.push_back(static_cast<char8_t>(asHex.at(i)));
507 }
508 }
509 else
510 {
511 serialized.push_back(unit);
512 }
513 }
514
515 serialized.push_back(u8'\"');
516 return serialized;
517 }
518
519 bJSON_MAKE_SERIALIZABLE_INLINE(JSONValue::ArrayType)
520 {
521 std::u8string serialized{u8"["};
522 for (auto first{true}; const auto &element : val)
523 {
524 if (element.type == JSONValue::JSONValueType::undefined)
525 {
526 continue;
527 }
528
529 if (!first)
530 {
531 serialized.push_back(u8',');
532 }
533 else
534 {
535 first = false;
536 }
537
538 serialized.push_back(u8' ');
539 serialized.append(serialize(element));
540 }
541 serialized.append(u8" ]");
542 return serialized;
543 };
544
545 bJSON_MAKE_SERIALIZABLE_INLINE(JSONValue::ObjectType)
546 {
547 std::u8string serialized{u8"{"};
548 for (auto first{true}; const auto &[key, value] : val)
549 {
550 if (value.type == JSONValue::JSONValueType::undefined)
551 {
552 continue;
553 }
554
555 if (!first)
556 {
557 serialized.push_back(u8',');
558 }
559 else
560 {
561 first = false;
562 }
563
564 serialized.push_back(u8' ');
565 serialized.append(serialize(JSONValue{key}));
566 serialized.append(u8" : ");
567 serialized.append(serialize(value));
568 }
569 serialized.append(u8" }");
570 return serialized;
571 }

573 inline bJSON_DEFINE_SERIALIZATION(JSONValue)
574 {
575 std::u8string serialized{u8""};
576 switch (val.type)
577 {
578 case JSONValue::JSONValueType::literal:
579 serialized.append(serialize(std::get<JSONValue::LiteralType>(val.value)));
580 break;
581 case JSONValue::JSONValueType::number:
582 serialized.append(serialize(std::get<JSONValue::NumberType>(val.value)));
583 break;
584 case JSONValue::JSONValueType::string:
585 serialized.append(serialize(std::get<JSONValue::StringType>(val.value)));
586 break;
587 case JSONValue::JSONValueType::array:
588 serialized.append(serialize(std::get<JSONValue::ArrayType>(val.value)));
589 break;
590 case JSONValue::JSONValueType::object:
591 serialized.append(serialize(std::get<JSONValue::ObjectType>(val.value)));
592 break;
593 case JSONValue::JSONValueType::undefined:
594 default:
595 throw std::exception{"JSONValue::type was 'undefined' -- it can not be serialized!"};
596 break;
597 }
598 return serialized;
599 }

757/* //
758// DO NOT REMOVE THIS SECTION! //
759// //
760// See the above macro usages for examples of how to serialize a type. //
761// //
762// As an additional example, consider the hypothetical struct which has the following definition: //
763// //
764// struct Example //
765// { //
766// const std::u8string name{u8""}; //
767// Example* const parent{nullptr}; //
768// }; //
769// //
770// To declare Example as json serializable and to define the serialization implementation, the helper macro is used:
771//
772// //
773// bJSON_MAKE_SERIALIZABLE(Example) //
774// { //
775// if (val.name.empty()) //
776// { //
777// throw std::exception{"Example::name was empty-- cannot serialize to JSON."}; //
778// } //
779// std::string serialized{u8R"""({ "name" : )"""}; //
780// serialized.append(serialize(val.name)); //
781// serialized.append(u8R"""(, "parent" : )"""); //
782// if (val.parent) //
783// { //
784// serialized.append(serialize(val.parent->name)); //
785// } //
786// else //
787// { //
788// serialized.append(serialize(JSONValue::LiteralType::null_v)); //
789// } //
790// serialized.append(u8" }"); //
791// return serialized; //
